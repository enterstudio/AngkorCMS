######################
#                    #
#    INSTALLATION    #
#                    #
######################


In command line, launch in order:
	"php artisan vendor:publish"
	"php artisan angkorcmsnews:install"

Register the service provider in 'config/app.php' :
		'AngkorCMS\News\AngkorCMSNewsServiceProvider',
